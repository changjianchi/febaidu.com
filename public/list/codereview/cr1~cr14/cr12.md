## 黎明：分享了require使用的方式差异以及常见问题解疑

### esl.js

```javascript

require('XXX');      // 模块XXX必须是已经define过的

define(['XXX'], function() {
    var AAA = require('AAA')     // 这种写法是错的
})

define(function() {
    var AAA = require('AAA')     // 正确的， 条件加载无效
});

require(['XXX'])    // 这种是运行时加载，任何地方都能用，可以条件加载

```
